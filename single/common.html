// 判断类型
function Type(obj) {
    var typeStr = Object.prototype.toString.call(obj).split(" ")[1];
    return typeStr.substr(0, typeStr.length - 1).toLowerCase();
}

// 截取想要的键值对
function GetQueryString(name, str, end) {
    var str = str || window.location.search;
    var reg = new RegExp('(^|\\?|&)' + name + '=([^&]*)(&|$' + (end?("|"+end):"") + ')');
    var r = str.match(reg);
    if (r != null) return decodeURIComponent(r[2]); return null;
}

// 日期加减
function DateAddDay(date, days) {
    var d = new Date(date);
    return new Date(d.setDate(d.getDate() + days));
}

// 区间内的随机数
function Random(min, max) {
    return (min||0) + Math.random() * ((max||1) - (min||0));
}

// 判断对象或数组为空
function isEmpty(e) {
    for (var t in e) return false;
    return true;
}

// 自动补零
function addZero(num, n) {
    var len = num.toString().length, n = n || 2;
    while (len < n) { num = '0' + num; len++; }
    return num + '';
}

// 产生随机颜色
function randomColor() {
    var rand = Math.floor(Math.random() * 0xFFFFFF).toString(16);
    if (rand.length == 6) {
        return '#' + rand;
    } else {
        return randomColor();
    }
}

// 生成唯一标识
function guid(len) {
    return Math.random().toString(36).substring(2, (len||5)+2);
}

// 将字符串转为唯一标识
function hashCode(str){
    var hash = 0;
    if (str.length == 0) return hash;
    for (i = 0; i < str.length; i++) {
        char = str.charCodeAt(i);
        hash = ((hash << 5) - hash) + char;
        hash = hash & hash;
    }
    return hash;
}

// 检测密码强度
function CheckPasswordLevel(str) {
    var level = 0;
    if (str.length < 6) return level;
    if (/[0-9]/.test(str)) level++;
    if (/[a-z]/.test(str)) level++;
    if (/[A-Z]/.test(str)) level++;
    if (/\.|-|_/.test(str)) level++;
    return level;
}

// 将 html 编译成 text，与其反编译
function encodeHTML(str) {
    var elem = document.createElement('span');
    elem.appendChild(document.createTextNode(str));
    return elem.innerHTML;
}
function decodeHTML(str) {
    str = str.replace(/</g, '<').replace(/>/g, '>')
    var elem = document.createElement('span');
    elem.innerHTML = str;
    return elem.textContent || elem.innerText;
}
