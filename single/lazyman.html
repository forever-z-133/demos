LazyMan 的实现（队列的应用）

普通队列的实现

function LazyMan(name) {
    this.tick = [];
    this.name = name || 'Man';
    var self = this;

    var fn = (function(){
        return function(){
            console.log('My name is ', this.name);
            this.next();
        }
    })(name);

    this.tick.push(fn);

    setTimeout(function(){
        self.next();
    }, 0);
}
LazyMan.prototype = {
    next: function() {
        var fn = this.tick.shift();
        fn && fn.call(this);
    },
    eat: function(name, time) {
        var self = this;
        var fn = (function(){
            return function(){
                console.log('eat ', self.name);
                self.next();
            }
        })(self.name, time);
        self.tick.push(fn);
        return self;
    },
    sleep: function(time, fisrt) {
        var self = this;
        var fn = (function(name, time, first){
            return function(){
                setTimeout(function(){
                    console.log('sleep ', (fisrt ? 'fisrt' : ''), time + 's', self.name);
                    self.next();
                }, time * 1000);
            }
        })(self.name, time, first);
        !fisrt ? self.tick.push(fn) : self.tick.unshift(fn);
        return self;
    },
    sleepFirst: function(time) {
        this.sleep(time, true);
        return this;
    }
};
