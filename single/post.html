var POSTING = false;
function Post() {
    // 请求未完成前不允许再次请求
    // 感觉以后应该改成队列，但队列后依旧很难判断重复请求
    if (POSTING) { return; } else POSTING = true;
    // 正式请求
    var P = $.post.apply(this, arguments);
    // 请求超时(注:调试时间也算在其中，所以应在发布时才激活此代码)
    var Timer = setTimeout(function () {
        _callback();
        alert('请求超过 5 秒了，看来出了点问题');
    }, 5000);
    // 请求成功
    P.done(function () {
        _callback();
    });
    // 请求失败
    P.fail(function () {
        _callback();
        alert('请求出错...');
    });
    // 请求完成
    function _callback() {
        P.abort();
        clearTimeout(Timer);
        POSTING = false;
    }
    return P;
}
