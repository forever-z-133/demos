// 获取此元素现在已有的样式
if (!HTMLElement.currentStyle) {
    function _getStyle(prop) {
        var _s = window.getComputedStyle(this, null)
        return prop ? _s[prop] : _s;
    }
    HTMLElement.prototype.currentStyle = _getStyle;
    HTMLElement.prototype.getStyle = _getStyle;
}
