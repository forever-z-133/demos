根据 别人博客 所写，按 Vue.js 的核心原理（数据劫持和订阅发布）来实现一个简陋的 MVVM 框架

能支持数据绑定，双向绑定，添加事件。暂不支持数组，列表或条件渲染。


<div id="mvvm-app">
  <input type="text" v-model="word">
  <p>{{word}}</p>
  <button v-on:click="sayHi">change model</button>
</div>
<script>
var vm = new MVVM({
  el: '#mvvm-app',
  data: {
     word: 'Hello World!'
  },
  computed: {
    'myName': function() {
      return [1];
    },
  },
  methods: {
    sayHi: function() {
       this.word = 'Hi, everybody!';
    }
  }
});
</script>

{{word}}

来自 computed 的：{{myName}}
